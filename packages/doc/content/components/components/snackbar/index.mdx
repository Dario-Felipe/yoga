# Snackbar

### Reference

Gympass `<Snackbar />` is the proper component to show alert messages.

The `Snackbar` may have an icon, a custom action and a close button.

### Usage

```javascript state
render(() => {
  const [isOpen, setIsOpen] = useState(false);

  const handleOpen = () => {
    setIsOpen(true);
  };

  const handleOnClose = () => {
    setIsOpen(false);
  };

  const handleOnAction = () => {
    alert('onAction function');
  };

  return (
    <>
      <Button onClick={handleOpen}>Open snackbar</Button>

      <Snackbar
        open={isOpen}
        message="We are processing your file."
        actionLabel="Action"
        onAction={handleOnAction}
        onClose={handleOnClose}
      />
    </>
  );
});
```

#### Variants

```javascript state
render(() => {
  const [isOpen, setIsOpen] = useState(false);

  const handleOpen = variant => {
    setIsOpen(state => ({ ...state, [variant]: true }));
  };

  const handleOnClose = variant => {
    setIsOpen(state => ({ ...state, [variant]: false }));
  };

  const handleOnAction = () => {
    alert('onAction function');
  };

  return (
    <>
      <section style={{ display: 'flex', justifyContent: 'space-between', width: 700 }}>
        <Button.Outline onClick={() => handleOpen('info')}>Open info snackbar</Button.Outline>
        <Button.Outline onClick={() => handleOpen('success')}>Open success snackbar</Button.Outline>
        <Button.Outline onClick={() => handleOpen('failure')}>Open failure snackbar</Button.Outline>
      </section>
      
      <Snackbar
        open={isOpen.info}
        message="We are processing your file."
        onClose={() => handleOnClose('info')}
      />

      <Snackbar
        open={isOpen.success}
        variant="success"
        message="Eligible added successfully."
        onClose={() => handleOnClose('success')}
      />

      <Snackbar
        open={isOpen.failure}
        variant="failure"
        message="Something went wrong, try again."
        actionLabel="Retry"
        onAction={handleOnAction}
        onClose={() => handleOnClose('failure')}
      />
    </>
  );
});
```
