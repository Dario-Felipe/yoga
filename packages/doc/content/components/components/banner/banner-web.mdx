## Usage

### Banner default

```javascript
  <Box
    display="flex"
    flex={1}
    justifyContent="space-between"
  >

    <Banner
      margin="small"
      variant='success'
      message='Success Banner'
    />

    <Banner
      margin="small"
      variant='informative'
      message='Informative Banner' 
    />

    <Banner
      margin="small"
      variant='attention'
      message='Attention Banner' 
    />
  </Box>
```

### Banner with action button

```javascript state
render(() => {
  const successBannerProps = {
    variant: 'success',
    message: 'Success Banner',
  };

  const informativeBannerProps = {
    variant: 'informative',
    message: 'Informative Banner',
  };

  const attentionBannerProps = {
    variant: 'attention',
    message: 'Attention Banner',
  };

  const handleOnAction = (message) => {
    alert(`This is your custom action ${message}`);
  };

  return (
    <Box
      display="flex"
      flexDirection="column"
      flex={1}
      justifyContent="space-between"
    >
      
      <Banner
        margin="small"
        variant={successBannerProps.variant}
        message={successBannerProps.message}
        actionLabel='Custom Action'
        onAction={() => handleOnAction(successBannerProps.message)}
      />

      <Banner
        margin="small"
        variant={informativeBannerProps.variant}
        message={informativeBannerProps.message}
        actionLabel='Custom Action'
        onAction={() => handleOnAction(informativeBannerProps.message)}
      />

      <Banner
        margin="small"
        variant={attentionBannerProps.variant}
        message={attentionBannerProps.message}
        actionLabel='Custom Action'
        onAction={() => handleOnAction(attentionBannerProps.message)}
      />
    </Box>
  );
});
```
