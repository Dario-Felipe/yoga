# Menu
```javascript state
  render(() => {
  const [isOpen, setIsOpen] = useState(false);

  const handleOpen = () => {
    setIsOpen(true);
  };

  const handleOnClose = () => {
    setIsOpen(false);
  };

  function showAlert() {
    alert('Click');
  } 

  return (
    <Box display="flex" flexDirection="column" alignItems="center">
      <Button onClick={handleOpen}>Open Menu</Button>

      <Menu isOpen={isOpen} onClose={handleOnClose}>
        <Menu.Item onClick={showAlert} icon={FlagBrazil}>
          Item 1
        </Menu.Item>
        <Menu.Item onClick={showAlert} icon={FlagUS}>
          Item 2
        </Menu.Item>
        <Menu.Item onClick={showAlert} icon={FlagPortugal}>
          Item 3
        </Menu.Item>
      </Menu>    
    </Box>
  );
});
```
