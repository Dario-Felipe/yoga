### Reference

Gympass `<List />` component implements all our design guidelines, be sure to wrap your items with `<List.Item/>` that uses default spacing, border and color pallete tokens.

If you need to use an anchor, `<List.LinkItem/>` is also available.

### Usage

```javascript state
const Wrapper = props => (
  <Box
    display="flex"
    justifyContent="space-between"
    {...props}
  />
);

render(() => {
  const [locker, toggleLocker] = useState(false);
  const [wifi, toggleWifi] = useState(false);
  const [parking, toggleParking] = useState(false);

  return (
    <List>
      <List.Item>
        <Wrapper>
          <span>Locker</span>
          <Checkbox.Switch
            checked={locker}
            onChange={() => toggleLocker(!locker)}
          />
        </Wrapper>
      </List.Item>

      <List.Item>
        <Wrapper>
          <span>Wi-fi</span>
          <Checkbox.Switch checked={wifi} onChange={() => toggleWifi(!wifi)} />
        </Wrapper>
      </List.Item>

      <List.Item>
        <Wrapper>
          <span>Parking</span>
          <Checkbox.Switch
            checked={parking}
            onChange={() => toggleParking(!parking)}
          />
        </Wrapper>
      </List.Item>

      <List.LinkItem
        href="https://www.gympass.com/"
        target="_blank"
      >
        <Text>Link</Text>
      </List.LinkItem>
    </List>
  );
});
```
