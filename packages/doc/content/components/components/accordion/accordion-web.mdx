### Usage

#### Simple

```javascript state
const List = styled.ul`
  padding-left: 20px;
  margin: 0 0 4px;
  display: flex;
  flex-direction: column;
  align-items: flex-start;
`;

const Item = styled.li`
  font-size: 10px;
`;


render(() => {
  return (
    <>
      <Accordion title="Title">
        <Accordion.Content>
          <Box d="flex" flexDirection="column" gap="32px">
            <Text color="deep">Subtitle with a list of some items:</Text>

            <List>
              <Item>
                <Text color="deep">Item 1</Text>
              </Item>

              <Item>
                <Text color="deep">Item 2</Text>
              </Item>

              <Item>
                <Text color="deep">Item 3</Text>
              </Item>

              <Item>
                <Text color="deep">Item 4</Text>
              </Item>

              <Item>
                <Text color="deep">Item 5</Text>
              </Item>
            </List>
          </Box>

          <Button small inverted>
            Small button
          </Button>
        </Accordion.Content>
      </Accordion>
    </>
  );
});
```
