## flex

```javascript state
const Stack = styled.div`
  width: 140px;
  height: 110px;
  border: 1px solid #ccc;
  padding: 4px;

  ${display}
  ${flexes}
`;

const Item = styled.div`
  background-color: lavender;
  padding: 4px;

  ${order}
`;

render(() => {
  return (
    <Stack
      d="flex"
      flexDirection="column"
      alignItems="center"
      justifyContent="space-around"
    >
      <Item order={0}>Making</Item>
      <Item order={2}>wellbeing</Item>
      <Item order={1}>universal</Item>
    </Stack>
  );
});
```
